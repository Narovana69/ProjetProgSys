format js:
const contextMenu = document.getElementById('context-menu');
const chatScroller = document.getElementById('chat-scroller');

// 1. Context Menu Right-Click Logic
document.addEventListener('contextmenu', (e) => {
    const msg = e.target.closest('.message-container');
    
    if (msg) {
        e.preventDefault();
        
        // Show and Position
        contextMenu.style.display = 'block';
        
        let x = e.clientX;
        let y = e.clientY;

        // Prevent overflow
        if (x + 190 > window.innerWidth) x -= 190;
        if (y + 150 > window.innerHeight) y -= 150;

        contextMenu.style.left = `${x}px`;
        contextMenu.style.top = `${y}px`;
        
        // Link message ID to menu
        contextMenu.dataset.targetId = msg.id;
    } else {
        contextMenu.style.display = 'none';
    }
});

// Close menu on click
document.addEventListener('click', (e) => {
    if (!contextMenu.contains(e.target)) {
        contextMenu.style.display = 'none';
    }
});

// 2. CRUD Actions
function deleteSelectedMessage(manualId = null) {
    const id = manualId || contextMenu.dataset.targetId;
    const el = document.getElementById(id);
    if (el) {
        el.remove();
        contextMenu.style.display = 'none';
    }
}

function copyMessageText() {
    const id = contextMenu.dataset.targetId;
    const text = document.querySelector(`#${id} .message-body`).innerText;
    navigator.clipboard.writeText(text);
    contextMenu.style.display = 'none';
}

// 3. Call Simulation
function startCall(type) {
    const color = type === 'video' ? '#5865f2' : '#23a55a';
    alert(`Starting ${type} call...`);
    // Example UI change
    document.querySelector('.chat-header').style.borderTop = `3px solid ${color}`;
}

// 4. Parameter/Settings Simulation
function openSettings() {
    alert("Opening Settings: User Profile, Privacy, Appearance...");
}
